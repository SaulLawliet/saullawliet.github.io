基本语法

数组

# 定义数组
array=(1 2 3)

# 数组长度
echo ${#array[@]}

# 按索引输出元素
echo ${array[0]}
echo ${array[-1]}

# 遍历数组
for value in ${array[@]}; do
  echo $value
done

代码片段

读文件

# 按空格, 制表符, 换行符等读取文件
for value in $(cat filename); do
  echo $value
done

#按行读取文件
while read line; do
  echo $line
done < filename

命令行参数相关

# 参数长度
echo $#

# 指定位置(第一个)
echo $1

# 遍历输出
for value in $@; do
  echo $value
done

输出脚本绝对路径

echo $(readlink -f $0)

输出脚本所在目录

echo $(dirname $(readlink -f $0))

输出上条命令执行结果

if (( $? )); then echo "failure"; else echo "success"; fi

切换用户执行

su other_user << !
  __command1__
  __command2__
  exit
!
